create table hero (
  hero_id int not null,
  name char(25) not null,
  kills int default 0,
  weapon_id int not null,
  primary key (hero_id),
  foreign key (weapon_id) references Weapon(weapon_id)
);

create table weapon (
  weapon_id int not null,
  name char(25) not null,
  kills int default 0,
  primary key (weapon_id)
);
